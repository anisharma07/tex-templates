\documentclass[aspectratio=169, 10pt]{beamer}
\usetheme{default}
\usecolortheme{default}

% Typography: Elegant Light
% Note: Standard LaTeX fonts don't always have a "light" weight easily accessible without XeLaTeX/LuaLaTeX
% We will simulate a lighter feel with color and font choice (Computer Modern Sans Serif is naturally quite light)
\usefonttheme{professionalfonts}
\usepackage[T1]{fontenc}
\usepackage{lmodern} % Latin Modern

% Packages
\usepackage{amsmath,amssymb,amsthm}
\usepackage{graphicx}
\usepackage{listings}
\usepackage{xcolor}

% Color scheme: Soft Grey & Teal
\definecolor{softteal}{RGB}{0, 128, 128}
\definecolor{lightgrey}{RGB}{80, 80, 80}

\setbeamercolor{frametitle}{fg=softteal,bg=white}
\setbeamerfont{frametitle}{series=\mdseries, size=\Large} % Medium series (not bold)
\setbeamercolor{title}{fg=softteal}
\setbeamerfont{title}{series=\mdseries, size=\huge}
\setbeamercolor{structure}{fg=softteal}
\setbeamercolor{normal text}{fg=lightgrey,bg=white}

% Unbold everything possible
\setbeamerfont{block title}{series=\mdseries}
\setbeamerfont{caption name}{series=\mdseries}
\setbeamerfont{description item}{series=\mdseries}

\setbeamercolor{block title}{bg=white, fg=softteal}
\setbeamercolor{block body}{bg=white} % Minimalist blocks (no background)
\setbeamertemplate{blocks}[rounded][shadow=false]

\setbeamertemplate{navigation symbols}{}
\setbeamertemplate{footline}[frame number]
