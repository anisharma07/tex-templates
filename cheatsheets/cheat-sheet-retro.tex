\sectiontitle{The 5 Notations}

\begin{tabular}{|c|l|c|}
\hline
\textbf{Not.} & \textbf{Meaning} & \textbf{Like} \\
\hline
$\Theta$ & Tight bound & $a = b$ \\
$O$ & Upper bound & $a \leq b$ \\
$\Omega$ & Lower bound & $a \geq b$ \\
$o$ & Strict upper & $a < b$ \\
$\omega$ & Strict lower & $a > b$ \\
\hline
\end{tabular}

\sectiontitle{Definitions}

\subsectiontitle{$\Theta(g(n))$}

$\exists c_1, c_2, n_0 > 0:$

$0 \leq c_1 g(n) \leq f(n) \leq c_2 g(n)$

$\forall n \geq n_0$

\subsectiontitle{$O(g(n))$}

$\exists c, n_0 > 0:$

$0 \leq f(n) \leq c \cdot g(n)$

$\forall n \geq n_0$

\subsectiontitle{$\Omega(g(n))$}

$\exists c, n_0 > 0:$

$0 \leq c \cdot g(n) \leq f(n)$

$\forall n \geq n_0$

\subsectiontitle{$o(g(n))$}

$\forall c > 0, \exists n_0:$

$0 \leq f(n) < c \cdot g(n)$

OR: $\lim_{n \to \infty} \frac{f(n)}{g(n)} = 0$

\subsectiontitle{$\omega(g(n))$}

$\forall c > 0, \exists n_0:$

$0 \leq c \cdot g(n) < f(n)$

OR: $\lim_{n \to \infty} \frac{f(n)}{g(n)} = \infty$

\sectiontitle{Key Theorems}

\textbf{[Th 3.1]} $f = \Theta(g) \iff f = O(g)$ AND $f = \Omega(g)$

\textbf{[Poly]} $p(n) = \sum_{i=0}^d a_i n^i \implies p(n) = \Theta(n^d)$

\textbf{[Transpose]} $f(n) = O(g(n)) \iff g(n) = \Omega(f(n))$

\sectiontitle{Growth Hierarchy}

$O(1) < O(\lg n) < O(\sqrt{n}) < O(n)$

$< O(n \lg n) < O(n^2) < O(n^3)$

$< O(2^n) < O(n!)$

\sectiontitle{Properties}

\textbf{[Transitivity]}

$f = \Theta(g), g = \Theta(h) \implies f = \Theta(h)$

\textbf{[Reflexivity]}

$f(n) = \Theta(f(n))$

\textbf{[Symmetry]}

$f(n) = \Theta(g(n)) \iff g(n) = \Theta(f(n))$

\columnbreak

\sectiontitle{Critical Limits}

\textbf{[Exp vs Poly]}

$\lim_{n \to \infty} \frac{n^b}{a^n} = 0$ $(a > 1)$

$\implies n^b = o(a^n)$

\textbf{[Poly vs Log]}

$\lim_{n \to \infty} \frac{\lg^b n}{n^a} = 0$ $(a > 0)$

$\implies \lg^b n = o(n^a)$

\textbf{[Factorial]}

$n! = \omega(2^n)$, $n! = o(n^n)$

\sectiontitle{Logarithms}

$\lg n = \log_2 n$, $\ln n = \log_e n$

\textbf{Identities:}
\begin{align*}
a &= b^{\log_b a}\\
\log(ab) &= \log a + \log b\\
\log(a^k) &= k \log a\\
\log_b a &= \frac{\log a}{\log b}\\
\log(1/a) &= -\log a
\end{align*}

\sectiontitle{Exponentials}

$a^0 = 1$, $a^1 = a$, $a^{-1} = 1/a$

$(a^m)^n = a^{mn}$, $a^m a^n = a^{m+n}$

\textbf{[Natural exp]}

$e^x \geq 1 + x$

$e^x = 1 + x + \Theta(x^2)$ as $x \to 0$

\sectiontitle{Factorials}

$n! = 1 \cdot 2 \cdot 3 \cdots n$

\textbf{[Stirling]}

$n! \approx \sqrt{2\pi n} (n/e)^n$

\textbf{[Log factorial]}

$\lg(n!) = \Theta(n \lg n)$

\sectiontitle{Floor/Ceiling}

$\lfloor x \rfloor$ :: max int $\leq x$

$\lceil x \rceil$ :: min int $\geq x$

$x - 1 < \lfloor x \rfloor \leq x \leq \lceil x \rceil < x + 1$

\sectiontitle{Modular Arithmetic}

$a \bmod n = a - n\lfloor a/n \rfloor$

$0 \leq a \bmod n < n$

$a \equiv b \pmod{n} \iff n | (b - a)$

\columnbreak

\sectiontitle{Sequences}

\textbf{[Fibonacci]}

$F_0=0, F_1=1, F_i=F_{i-1}+F_{i-2}$

$F_i \approx \phi^i / \sqrt{5}$ where $\phi \approx 1.618$

\textbf{[Iterated Log]}

$\lg^* n$ : grows very slowly.

$2,4,16,65536 \to 1,2,3,4$

\sectiontitle{Loop Analysis}

\textbf{Single loop:} $O(n)$

\textbf{Nested (both n):} $O(n^2)$

\textbf{Dependent:} $\sum i \to O(n^2)$

\textbf{Halving:} $n=n/2 \to O(\lg n)$

\textbf{Tree recursion:} $O(n \lg n)$

\sectiontitle{Summation}

$\sum_{i=1}^n 1 = n$

$\sum_{i=1}^n i = \Theta(n^2)$

$\sum_{i=1}^n i^2 = \Theta(n^3)$

$\sum_{i=0}^n a^i = \frac{a^{n+1}-1}{a-1}$

\sectiontitle{Practical Limits}

\textit{Max n for 1 sec:}

$O(n) \to 10^8$

$O(n \lg n) \to 10^7$

$O(n^2) \to 10^4$

$O(n^3) \to 500$

$O(2^n) \to 25$

$O(n!) \to 11$

\sectiontitle{Quick Rules}

\textbf{Polynomials:} Drop lower terms

\textbf{Nested loops:} Multiply

\textbf{Sequential:} Add (Take Max)

\textbf{Recursive:} Master Thm

\sectiontitle{Common Mistakes}

[ERR] "at least O(n)"

[ERR] $2^n$ vs $n^2$

[ERR] Counting input in space

\sectiontitle{Space Complexity}

Count only aux space.

\textbf{Recursion:} $O(depth)$

\sectiontitle{Master Theorem}

$T(n) = aT(n/b) + f(n)$

$c_{crit} = \log_b a$

\textbf{[Case 1]} $f(n)$ small

$\implies T(n) = \Theta(n^{c_{crit}})$

\textbf{[Case 2]} $f(n)$ equal

$\implies T(n) = \Theta(n^{c_{crit}} \lg n)$

\textbf{[Case 3]} $f(n)$ large

$\implies T(n) = \Theta(f(n))$

\sectiontitle{Proving Bounds}

\textbf{[Theta]} Find $c_1, c_2, n_0$

\textbf{[Big-O]} Find $c, n_0$ OR limit $<\infty$

\textbf{[Omega]} Find $c, n_0$ OR limit $>0$
