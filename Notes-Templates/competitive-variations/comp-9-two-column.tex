\section{Graph Algorithms}
\subsection{DFS/BFS}
Standard traversals.

\begin{lstlisting}[language=C++]
void dfs(int u) {
    vis[u] = 1;
    for(int v : adj[u]) if(!vis[v]) dfs(v);
}
\end{lstlisting}

\begin{optimizebox}
Use iterative for deep recursion.
\end{optimizebox}
